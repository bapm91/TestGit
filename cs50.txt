__________________________________знак процента замещает значение стоящее после строки соответственно символу
%i - замена целых чисел
%c - замена char
%f - замена float и double (иногда они взаимозаменяемые)
%lld - замена long long
%s - замена string
и т.д.

_______________________________другие управляющие последовательности__________________________
\n - переход на новую строку
\r - 
\t - отступ равный табуляции
\' -
\" - вставка кавычек в строку 
\\ -
\0 -
и т.д.

_______________________________get'еры_________________________________________________________
применимы ко всем примитивным типам данных, например getInt();
считывают ввод из консоли.

_______________________________некоторые команды для командной строки(терминала IDE cs50)_______________
cd - 			перейти в корневую папку
ls - 			показать все папки и файлы в текущей директории (сокращение от list)
Ctrl + L - 		очистить терминал (история команд сохраняется в CS50 IDE)
cd (нужная папка) - 	перейти в нужную папку
cd .. - 		перейти в папку выше по иерархии (ближе к корневой)
make (file name) - 	скомпилировать файл (писать без расширения)
./(file name) - 	запустить исполняемый файл (писать без расширения)
mv source destination - переименовать файл, где source — текущее имя файла, destination — новое имя файла

_______________________________библиотеки_______________________________________________________
less /usr/include/*.h - просмотреть исходники библиотеки
#include <> - 		включить в разработку
stdio.h - 		библиотека содержит функции ввода/вывода
string.h - 		содержит операции со строками (определение длины, сложение и прочее)

_______________________________линукс команды в консоли_________________________________________
ls -a  - 		показывает и скрытые файлы дериктории
pwd  - 			показывает текущее положение в директориях
cd ./ or cd  -		перейти в нужную папку в текущей директории
cd ~/  -		перейти в корневую директорию 		
ls -la  - 		показывает таблицу
ls  - 			показвает содержимое директории
touch (file name.*)  -  создать файл
nano (/file name.*)  - 	редактировать файл
mkdir (folder name)  - 	создать папку
rm -rf /  - 		удалить все
rm ./(file name)  -	удалить файл
rm -r ./(name folder) - удалить папку
mv (от куда) (куда)  -  переместить
sudo  - 		доступ от имени пользователя
sudo apt-get install (lib name)-установка библиотеки
Tab Tab  - 		подсказки
sudo apt-get install update - проверка обновлений линукса в данных
sudo apt-get install upgrate - скачивание и установка обновлений
sudo pppoeconf - 	настройка нового подключения через PPPOE соединение
sudo poff dsl-provider - разорвать соединение через PPPOE(отключить)
sudo pon dsl-provider -	подключиться с существующими настройками к интернету
